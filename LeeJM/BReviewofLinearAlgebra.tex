\begin{enumerate}
\item[(i)] \textsc{Positivity} $|v| \geq 0$, \, $\forall \, v \in V$, $|v| =0$ iff $v=0$ 
\item[(ii)] \textsc{Homogeneity} $|cv| = |c| |v|$ \quad \, $\forall \, c \in \mathbb{R}$, $v\in V$
\item[(iii)] \textsc{Triangle Inequality} $|v+w | \leq |v| + |w|$, \, $\forall \, v,w \in V$
\end{enumerate}
